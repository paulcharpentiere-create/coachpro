// Storage adapter - fonctionne dans le navigateur avec localStorage const storage = { async get(key) { try { const val = localStorage.getItem(key); return val ? { value: val } : null; } catch { return null; } }, async set(key, value) { try { localStorage.setItem(key, value); return { key, value }; } catch { return null; } }, async delete(key) { try { localStorage.removeItem(key); return { key, deleted: true }; } catch { return null; } } }; export default storage;
